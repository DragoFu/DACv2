001package fr.aumgn.dac2;
002
003import org.bukkit.Bukkit;
004import org.bukkit.plugin.java.JavaPlugin;
005
006import com.google.gson.FieldNamingPolicy;
007import com.google.gson.Gson;
008import com.google.gson.GsonBuilder;
009
010import fr.aumgn.bukkitutils.command.CommandsRegistration;
011import fr.aumgn.bukkitutils.command.arg.CommandArgFactory;
012import fr.aumgn.bukkitutils.gson.GsonLoadException;
013import fr.aumgn.bukkitutils.gson.GsonLoader;
014import fr.aumgn.bukkitutils.gson.typeadapter.DirectionTypeAdapterFactory;
015import fr.aumgn.dac2.arena.Arena;
016import fr.aumgn.dac2.arena.regions.GsonRegionFactory;
017import fr.aumgn.dac2.commands.AdminCommands;
018import fr.aumgn.dac2.commands.ArenasCommands;
019import fr.aumgn.dac2.commands.SetupCommands;
020import fr.aumgn.dac2.commands.SpectatorCommands;
021import fr.aumgn.dac2.commands.StageCommands;
022import fr.aumgn.dac2.commands.arg.ArenaArg;
023import fr.aumgn.dac2.commands.arg.StageArg;
024import fr.aumgn.dac2.commands.worldedit.DisabledWorldEditCommands;
025import fr.aumgn.dac2.commands.worldedit.SelectCommands;
026import fr.aumgn.dac2.commands.worldedit.SetupWECommands;
027import fr.aumgn.dac2.config.DACConfig;
028import fr.aumgn.dac2.stage.Stage;
029
030public class DACPlugin extends JavaPlugin {
031
032    private final static double GSON_VERSION = 1.0;
033
034    private DAC dac;
035
036    @Override
037    public void onEnable() {
038        dac = new DAC(this);
039
040        CommandArgFactory.register(Arena.class, new ArenaArg.Factory(dac));
041        CommandArgFactory.register(Stage.class, new StageArg.Factory(dac));
042        CommandsRegistration registration = new CommandsRegistration(
043                this, dac.getConfig().getLocale());
044        registration.register(new AdminCommands(dac));
045        registration.register(new ArenasCommands(dac));
046        registration.register(new SetupCommands(dac));
047        if (isWorldEditEnabled()) {
048            registration.register(new SetupWECommands(dac));
049            registration.register(new SelectCommands(dac));
050        } else {
051            registration.register(new DisabledWorldEditCommands(dac));
052        }
053        registration.register(new StageCommands(dac));
054        registration.register(new SpectatorCommands(dac));
055
056        getLogger().info("Enabled.");
057    }
058
059    public boolean isWorldEditEnabled() {
060        return Bukkit.getPluginManager().isPluginEnabled("WorldEdit");
061    }
062
063    @Override
064    public void onDisable() {
065        getLogger().info("Disabled.");
066    }
067
068    public GsonLoader getGsonLoader() {
069        Gson gson = new GsonBuilder()
070            .registerTypeAdapterFactory(new DirectionTypeAdapterFactory())
071            .registerTypeAdapterFactory(new GsonRegionFactory())
072            .setVersion(GSON_VERSION)
073            .setFieldNamingPolicy(FieldNamingPolicy.LOWER_CASE_WITH_DASHES)
074            .setPrettyPrinting()
075            .create();
076        return new GsonLoader(gson, this);
077    }
078
079    public DACConfig reloadDACConfig() {
080        try {
081            GsonLoader loader = getGsonLoader();
082            return loader.loadOrCreate("config.json", DACConfig.class);
083         } catch (GsonLoadException exc) {
084             getLogger().warning(
085                     "Unable to load configuration file.");
086             getLogger().warning("Using default values instead.");
087             return new DACConfig();
088         }
089    }
090}
