001package fr.aumgn.dac2.stage.join;
002
003import java.util.List;
004import java.util.Map;
005import java.util.Map.Entry;
006import java.util.Set;
007
008import org.bukkit.ChatColor;
009import org.bukkit.entity.Player;
010import org.bukkit.event.Listener;
011
012import fr.aumgn.bukkitutils.localization.PluginMessages;
013import fr.aumgn.bukkitutils.playerref.PlayerRef;
014import fr.aumgn.bukkitutils.playerref.map.PlayersRefHashMap;
015import fr.aumgn.bukkitutils.playerref.map.PlayersRefMap;
016import fr.aumgn.bukkitutils.playerref.set.PlayersRefHashSet;
017import fr.aumgn.bukkitutils.playerref.set.PlayersRefSet;
018import fr.aumgn.bukkitutils.util.Util;
019import fr.aumgn.dac2.DAC;
020import fr.aumgn.dac2.arena.Arena;
021import fr.aumgn.dac2.config.Color;
022import fr.aumgn.dac2.game.start.GameStartData;
023import fr.aumgn.dac2.game.start.PlayerStartData;
024import fr.aumgn.dac2.game.start.SimplePlayerStartData;
025import fr.aumgn.dac2.stage.Stage;
026
027public class JoinStage implements Stage, Listener, GameStartData {
028
029    private final DAC dac;
030    private final Arena arena;
031    private final Map<String, Color> colors;
032    private final PlayersRefMap<PlayerStartData> players;
033    private final PlayersRefSet spectators;
034
035    public JoinStage(DAC dac, Arena arena) {
036        this.dac = dac;
037        this.arena = arena;
038        this.colors = dac.getColors().toMap();
039        this.players = new PlayersRefHashMap<PlayerStartData>();
040        this.spectators = new PlayersRefHashSet();
041    }
042
043    @Override
044    public Arena getArena() {
045        return arena;
046    }
047
048    @Override
049    public void start() {
050        PluginMessages messages = dac.getMessages();
051        Util.broadcast("dac2.game.watch", 
052                messages.get("joinstage.start1", arena.getName()));
053        Util.broadcast("dac2.game.watch", messages.get("joinstage.start2"));
054    }
055
056    @Override
057    public void stop(boolean force) {
058        if (force) {
059            sendMessage(dac.getMessages().get("joinstage.stopped"));
060        }
061    }
062
063    @Override
064    public Listener[] getListeners() {
065        return new Listener[0];
066    }
067
068    @Override
069    public boolean contains(Player player) {
070        return players.containsKey(player);
071    }
072
073    @Override
074    public void sendMessage(String message) {
075        for (Player player : players.playersSet()) {
076            player.sendMessage(message);
077        }
078
079        String spectatorMessage = dac.getConfig().getSpectatorsMsg()
080                .format(new String[] { arena.getName(), message });
081        for (Player spectator : spectators.players()) {
082            spectator.sendMessage(spectatorMessage);
083        }
084    }
085
086    public void addPlayer(Player player, List<String> colorsName) {
087        Color color = null;
088        for (String rawColorName : colorsName) {
089            String colorName = rawColorName.toLowerCase();
090            if (colors.containsKey(colorName)) {
091                color = colors.get(colorName);
092                break;
093            }
094        }
095
096        addPlayer(player, color);
097    }
098
099    public void addPlayer(Player player, Color color) {
100        if (color == null || !colors.containsValue(color)) {
101            // Get first color available
102            color = colors.values().iterator().next();
103        }
104
105        PluginMessages msgs = dac.getMessages();
106        String playerName = color.chat + player.getDisplayName();
107        sendMessage(msgs.get("joinstage.join", playerName));
108
109        players.put(player, new SimplePlayerStartData(color, player));
110        colors.remove(color.name);
111
112        player.sendMessage(msgs.get("joinstage.playerslist"));
113        for (Entry<PlayerRef, PlayerStartData> playerIG : players.entrySet()) {
114            PlayerRef playerRef = playerIG.getKey();
115            PlayerStartData data = playerIG.getValue();
116
117            String name;
118            if (playerRef.isOnline()) {
119                name = data.getColor().chat
120                        + playerIG.getKey().getPlayer().getDisplayName();
121            } else {
122                name = ChatColor.ITALIC.toString() + data.getColor().chat
123                        + playerRef.getName();
124            }
125
126            player.sendMessage(msgs.get("joinstage.playerentry", name));
127        }
128    }
129
130    @Override
131    public boolean isSpectator(Player player) {
132        return spectators.contains(player);
133    }
134
135    @Override
136    public void addSpectator(Player player) {
137        spectators.add(player);
138    }
139
140    @Override
141    public void removeSpectator(Player player) {
142        spectators.remove(player);
143    }
144
145    public int size() {
146        return players.size();
147    }
148
149    @Override
150    public Map<PlayerRef, PlayerStartData> getPlayersData() {
151        return players;
152    }
153
154    @Override
155    public Set<PlayerRef> getSpectators() {
156        return spectators;
157    }
158}
