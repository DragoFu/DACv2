001package fr.aumgn.dac2.shape;
002
003import java.util.Iterator;
004import java.util.Set;
005
006import fr.aumgn.bukkitutils.geom.Vector;
007import fr.aumgn.bukkitutils.geom.Vector2D;
008import fr.aumgn.dac2.shape.column.Column;
009
010@ShapeName("arbitrary")
011public class ArbitraryFlatShape implements FlatShape {
012
013    private final Set<Vector2D> points;
014    private final double minY;
015    private final double maxY;
016
017    private transient Vector2D min2D;
018    private transient Vector2D max2D;
019
020    public ArbitraryFlatShape(Set<Vector2D> points, double minY, double maxY) {
021        this.points = points;
022        this.minY = minY;
023        this.maxY = maxY;
024
025        this.min2D = null;
026        this.max2D = null;
027    }
028
029    @Override
030    public boolean contains(Vector pt) {
031        return pt.getY() >= minY && pt.getY() <= maxY
032                && contains2D(pt.to2D());
033    }
034
035    @Override
036    public Iterator<Column> iterator() {
037        final Iterator<Vector2D> it = points.iterator();
038        return new Iterator<Column>() {
039
040            @Override
041            public boolean hasNext() {
042                return it.hasNext();
043            }
044
045            @Override
046            public Column next() {
047                Vector2D next = it.next();
048                return new Column(ArbitraryFlatShape.this, next);
049            }
050
051            @Override
052            public void remove() {
053                throw new UnsupportedOperationException();
054            }
055        };
056    }
057
058    @Override
059    public double getMinY() {
060        return minY;
061    }
062
063    @Override
064    public double getMaxY() {
065        return maxY;
066    }
067
068    @Override
069    public Vector2D getMin2D() {
070        if (min2D == null) {
071            calculateMinMax2D();
072        }
073
074        return min2D;
075    }
076
077    @Override
078    public Vector2D getMax2D() {
079        if (max2D == null) {
080            calculateMinMax2D();
081        }
082
083        return max2D;
084    }
085
086    private void calculateMinMax2D() {
087        double minX = Double.MAX_VALUE;
088        double minZ = Double.MAX_VALUE;
089        double maxX = -Double.MAX_VALUE;
090        double maxZ = -Double.MAX_VALUE;
091
092        for (Vector2D pt : points) {
093            if (minX > pt.getX()) {
094                minX = pt.getX();
095            }
096            if (minZ > pt.getZ()) {
097                minZ = pt.getZ();
098            }
099            if (maxX < pt.getX()) {
100                maxX = pt.getX();
101            }
102            if (maxZ < pt.getZ()) {
103                maxZ = pt.getZ();
104            }
105        }
106
107        min2D = new Vector2D(minX, minZ);
108        max2D = new Vector2D(maxX, maxZ);
109    }
110
111    @Override
112    public boolean contains2D(Vector2D pt) {
113        if (min2D == null) {
114            calculateMinMax2D();
115        }
116
117        return pt.isInside(min2D, max2D) && points.contains(pt);
118    }
119
120    @Override
121    public Column getColumn(Vector2D pt) {
122        return new Column(this, pt);
123    }
124}
