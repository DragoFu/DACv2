001package fr.aumgn.dac2.commands.arg;
002
003import org.bukkit.command.CommandSender;
004import org.bukkit.entity.Player;
005
006import fr.aumgn.bukkitutils.command.CommandsMessages;
007import fr.aumgn.bukkitutils.command.arg.CommandArg;
008import fr.aumgn.bukkitutils.command.arg.CommandArgFactory;
009import fr.aumgn.bukkitutils.command.exception.CommandError;
010import fr.aumgn.bukkitutils.command.exception.CommandUsageError;
011
012import fr.aumgn.dac2.DAC;
013import fr.aumgn.dac2.arena.Arena;
014import fr.aumgn.dac2.stage.Stage;
015
016public class StageArg extends CommandArg<Stage> {
017
018    public static class Factory extends CommandArgFactory<Stage> {
019
020        private final DAC dac;
021
022        public Factory(DAC dac) {
023            this.dac = dac;
024        }
025
026        @Override
027        public CommandArg<Stage> createCommandArg(CommandsMessages messages,
028                String string) {
029            return new StageArg(dac, messages, string);
030        }
031    }
032
033    public static class NoStageForArena extends CommandError {
034
035        private static final long serialVersionUID = 8898579114644791040L;
036
037        public NoStageForArena(DAC dac, Arena arena) {
038            super(dac.getCmdMessages()
039                    .get("stage.arg.nostageforarena", arena.getName()));
040        }
041    }
042
043    public static class NoStageForPlayer extends CommandError {
044
045        private static final long serialVersionUID = 2665340229475254801L;
046
047        public NoStageForPlayer(DAC dac) {
048            super(dac.getCmdMessages()
049                    .get("stage.arg.nostageforplayer"));
050        }
051    }
052
053    private final DAC dac;
054
055    public StageArg(DAC dac, CommandsMessages messages, String string) {
056        super(messages, string);
057        this.dac = dac;
058    }
059
060    @Override
061    public Stage value() {
062        if (!dac.getArenas().has(string)) {
063            throw new ArenaArg.NoSuchArena(dac, string);
064        }
065
066        Arena arena = dac.getArenas().get(string);
067        Stage stage = dac.getStages().get(arena);
068        if (stage == null) {
069            throw new NoStageForArena(dac, arena);
070        }
071
072        return stage;
073    }
074
075    @Override
076    protected Stage defaultFor(CommandSender sender) {
077        if (!(sender instanceof Player)) {
078            throw new CommandUsageError(dac.getCmdMessages()
079                    .get("stage.arg.needed"));
080        }
081
082        Stage stage = dac.getStages().get((Player) sender);
083        if (stage == null) {
084            throw new NoStageForPlayer(dac);
085        }
086
087        return stage;
088    }
089}
