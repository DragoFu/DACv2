001package fr.aumgn.dac2.commands.arg;
002
003import org.bukkit.command.CommandSender;
004import org.bukkit.entity.Player;
005
006import fr.aumgn.bukkitutils.command.CommandsMessages;
007import fr.aumgn.bukkitutils.command.arg.CommandArg;
008import fr.aumgn.bukkitutils.command.arg.CommandArgFactory;
009import fr.aumgn.bukkitutils.command.exception.CommandError;
010import fr.aumgn.bukkitutils.command.exception.CommandUsageError;
011
012import fr.aumgn.dac2.DAC;
013import fr.aumgn.dac2.arena.Arena;
014
015public class ArenaArg extends CommandArg<Arena> {
016
017    public static class Factory extends CommandArgFactory<Arena> {
018
019        private final DAC dac;
020
021        public Factory(DAC dac) {
022            this.dac = dac;
023        }
024
025        @Override
026        public CommandArg<Arena> createCommandArg(CommandsMessages messages,
027                String string) {
028            return new ArenaArg(dac, messages, string);
029        }
030    }
031
032    public static class NoSuchArena extends CommandError {
033        private static final long serialVersionUID = -4832133406864970323L;
034
035        public NoSuchArena(DAC dac, String name) {
036            super(dac.getCmdMessages().get("arena.arg.notfound", name));
037        }
038    }
039
040    public static class NotInArena extends CommandError {
041
042        private static final long serialVersionUID = 6112644121244362679L;
043
044        public NotInArena(DAC dac) {
045            super(dac.getCmdMessages().get("arena.arg.notinarena"));
046        }
047    }
048
049    private final DAC dac;
050
051    public ArenaArg(DAC dac, CommandsMessages messages, String string) {
052        super(messages, string);
053        this.dac = dac;
054    }
055
056    @Override
057    public Arena value() {
058        if (!dac.getArenas().has(string)) {
059            throw new NoSuchArena(dac, string);
060        }
061
062        return dac.getArenas().get(string);
063    }
064
065    @Override
066    protected Arena defaultFor(CommandSender sender) {
067        if (!(sender instanceof Player)) {
068            throw new CommandUsageError(dac.getCmdMessages()
069                    .get("arena.arg.needed"));
070        }
071
072        Arena arena = dac.getArenas().get((Player) sender);
073        if (arena == null) {
074            throw new NotInArena(dac);
075        }
076
077        return arena;
078    }
079}
