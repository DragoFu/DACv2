001package fr.aumgn.dac2.commands;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.bukkit.entity.Player;
007
008import fr.aumgn.bukkitutils.command.Command;
009import fr.aumgn.bukkitutils.command.NestedCommands;
010import fr.aumgn.bukkitutils.command.args.CommandArgs;
011import fr.aumgn.bukkitutils.command.exception.CommandError;
012import fr.aumgn.dac2.DAC;
013import fr.aumgn.dac2.arena.Arena;
014import fr.aumgn.dac2.config.Color;
015import fr.aumgn.dac2.stage.Stage;
016import fr.aumgn.dac2.stage.join.JoinStage;
017
018@NestedCommands("dac2")
019public class PlayerCommands extends DACCommands {
020
021    public PlayerCommands(DAC dac) {
022        super(dac);
023    }
024
025    @Command(name = "join", min = 0, max = -1, argsFlags = "a")
026    public void join(Player sender, CommandArgs args) {
027        if (dac.getStages().get(sender) != null) {
028            throw new CommandError(msg("join.alreadyingame"));
029        }
030
031        Arena arena = args.get('a', Arena)
032                .valueWithPermOr("dac2.stages.join.arena", sender);
033        Stage stage = dac.getStages().get(arena);
034
035        if (!(stage instanceof JoinStage)) {
036            throw new CommandError(msg("join.notjoinable"));
037        }
038
039        List<Color> colors = new ArrayList<Color>();
040        for (int i = 1; i < args.length(); i++) {
041            colors.add(args.get(i, Color).value());
042        }
043        ((JoinStage) stage).addPlayer(sender, colors);
044    }
045
046    @Command(name = "quit")
047    public void quit(Player sender) {
048        Stage stage = dac.getStages().get(sender);
049        if (stage == null) {
050            throw new CommandError(msg("quit.notingame"));
051        }
052
053        stage.onQuit(sender);
054    }
055}
