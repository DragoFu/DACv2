001package fr.aumgn.dac2.commands.worldedit;
002
003import org.bukkit.entity.Player;
004
005import com.sk89q.worldedit.LocalSession;
006import com.sk89q.worldedit.LocalWorld;
007import com.sk89q.worldedit.bukkit.BukkitUtil;
008import com.sk89q.worldedit.bukkit.WorldEditPlugin;
009import com.sk89q.worldedit.regions.RegionSelector;
010
011import fr.aumgn.bukkitutils.command.Command;
012import fr.aumgn.bukkitutils.command.NestedCommands;
013import fr.aumgn.bukkitutils.command.args.CommandArgs;
014import fr.aumgn.dac2.DAC;
015import fr.aumgn.dac2.arena.Arena;
016import fr.aumgn.dac2.arena.regions.Region;
017import fr.aumgn.dac2.shape.Shape;
018import fr.aumgn.dac2.shape.worldedit.WESelector;
019
020@NestedCommands({ "dac2", "select" })
021public class SelectCommands extends WorldEditCommands {
022
023    public SelectCommands(DAC dac) {
024        super(dac);
025    }
026
027    @Command(name = "pool", min = 1, max = 1, argsFlags = "s")
028    public void pool(Player sender, CommandArgs args) {
029        Arena arena = args.get(0, Arena).value();
030        WESelector selectorType = args.get('s', WESelector.class)
031                .valueOr(WESelector.Default);
032
033        setSelection(sender, arena, selectorType, arena.safeGetPool(dac));
034        sender.sendMessage(msg("select.pool.success"));
035    }
036
037    @Command(name = "start", min = 1, max = 1, argsFlags = "s")
038    public void start(Player sender, CommandArgs args) {
039        Arena arena = args.get(0, Arena).value();
040        WESelector selectorType = args.get('s', WESelector.class)
041                .valueOr(WESelector.Default);
042
043        setSelection(sender, arena, selectorType,
044                arena.safeGetStartRegion(dac));
045        sender.sendMessage(msg("select.start.success"));
046    }
047
048    @Command(name = "surrounding", min = 1, max = 1, argsFlags = "s")
049    public void surrounding(Player sender, CommandArgs args) {
050        Arena arena = args.get(0, Arena).value();
051        WESelector selectorType = args.get('s', WESelector.class)
052                .valueOr(WESelector.Default);
053
054        setSelection(sender, arena, selectorType,
055                arena.safeGetSurroundingRegion(dac));
056        sender.sendMessage(msg("select.surrounding.success"));
057    }
058
059    private void setSelection(Player sender, Arena arena,
060            WESelector selectorType, Region region) {
061        WorldEditPlugin worldEdit = getWorldEdit();
062        LocalWorld world = BukkitUtil.getLocalWorld(arena.getWorld());
063        Shape shape = region.getShape();
064
065        RegionSelector sel = selectorType.create(dac, world, shape);
066        LocalSession session = worldEdit.getSession(sender);
067        session.setRegionSelector(world, sel);
068        session.dispatchCUISelection(worldEdit.wrapPlayer(sender));
069    }
070}
