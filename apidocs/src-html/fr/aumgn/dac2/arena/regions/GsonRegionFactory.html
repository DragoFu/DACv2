001package fr.aumgn.dac2.arena.regions;
002
003import java.io.IOException;
004import java.lang.reflect.Constructor;
005import java.lang.reflect.InvocationTargetException;
006import java.util.HashMap;
007import java.util.Map;
008
009import com.google.gson.Gson;
010import com.google.gson.TypeAdapter;
011import com.google.gson.TypeAdapterFactory;
012import com.google.gson.reflect.TypeToken;
013import com.google.gson.stream.JsonReader;
014import com.google.gson.stream.JsonToken;
015import com.google.gson.stream.JsonWriter;
016
017import fr.aumgn.dac2.shape.ArbitraryFlatShape;
018import fr.aumgn.dac2.shape.CuboidShape;
019import fr.aumgn.dac2.shape.CylinderShape;
020import fr.aumgn.dac2.shape.EllipsoidShape;
021import fr.aumgn.dac2.shape.PolygonalShape;
022import fr.aumgn.dac2.shape.Shape;
023import fr.aumgn.dac2.shape.ShapeName;
024
025public class GsonRegionFactory implements TypeAdapterFactory {
026
027    private static final Map<String, Class<? extends Shape>> classes =
028            new HashMap<String, Class<? extends Shape>>();
029
030    public static void registerShape(Class<? extends Shape> shapeClass) {
031        ShapeName annotation =
032                shapeClass.getAnnotation(ShapeName.class);
033        if (annotation == null) {
034            return;
035        }
036
037        classes.put(annotation.value(), shapeClass);
038    }
039
040    static {
041        registerShape(CuboidShape.class);
042        registerShape(CylinderShape.class);
043        registerShape(PolygonalShape.class);
044        registerShape(EllipsoidShape.class);
045        registerShape(ArbitraryFlatShape.class);
046    }
047
048    private static class RegionTypeAdapter extends TypeAdapter<Region> {
049
050        private final Gson gson;
051        private final Constructor<Region> ctor;
052
053        public RegionTypeAdapter(Gson gson, Constructor<Region> ctor) {
054            this.gson = gson;
055            this.ctor = ctor;
056        }
057
058        @Override
059        public Region read(JsonReader reader) throws IOException {
060            reader.beginObject();
061
062            reader.nextName();
063            String type = reader.nextString();
064            Class<? extends Shape> clazz = classes.get(type);
065            if (clazz == null) {
066                while (reader.peek() != JsonToken.END_OBJECT) {
067                    reader.nextName();
068                    reader.skipValue();
069                }
070                reader.endObject();
071
072                return null;
073            }
074
075            reader.nextName();
076            Shape shape = gson.fromJson(reader, clazz);
077            reader.endObject();
078
079            try {
080                return ctor.newInstance(shape);
081            } catch (IllegalArgumentException _) {
082            } catch (InstantiationException _) {
083            } catch (IllegalAccessException _) {
084            } catch (InvocationTargetException _) {
085            }
086
087            return null;
088        }
089
090        @Override
091        public void write(JsonWriter writer, Region region)
092                throws IOException {
093            Shape shape = region.getShape();
094            ShapeName annotation = shape.getClass()
095                    .getAnnotation(ShapeName.class);
096            if (annotation == null) {
097                writer.nullValue();
098                return;
099            }
100            String shapeName = annotation.value();
101
102
103            writer.beginObject();
104            writer.name("shape");
105            writer.value(shapeName);
106            writer.name("data");
107            gson.toJson(shape, shape.getClass(), writer);
108            writer.endObject();
109        }
110    }
111
112    @SuppressWarnings("unchecked")
113    @Override
114    public <T> TypeAdapter<T> create(Gson gson, TypeToken<T> type) {
115        if (!Region.class.isAssignableFrom(type.getRawType())) {
116            return null;
117        }
118
119        try {
120            Constructor<Region> ctor = (Constructor<Region>)
121                    type.getRawType().getDeclaredConstructor(Shape.class);
122            ctor.setAccessible(true);
123            return (TypeAdapter<T>) new RegionTypeAdapter(gson, ctor)
124                    .nullSafe();
125        } catch (SecurityException _) {
126        } catch (NoSuchMethodException _) {
127        }
128
129        return null;
130    }
131}
