001package fr.aumgn.dac2.game.start;
002
003import java.util.Iterator;
004
005import org.bukkit.Bukkit;
006import org.bukkit.entity.Player;
007
008import fr.aumgn.bukkitutils.playerref.map.PlayersRefHashMap;
009import fr.aumgn.bukkitutils.playerref.map.PlayersRefMap;
010import fr.aumgn.bukkitutils.playerref.set.PlayersRefHashSet;
011import fr.aumgn.bukkitutils.playerref.set.PlayersRefSet;
012import fr.aumgn.dac2.DAC;
013import fr.aumgn.dac2.arena.Arena;
014import fr.aumgn.dac2.arena.regions.StartRegion;
015import fr.aumgn.dac2.config.Color;
016import fr.aumgn.dac2.exceptions.TooManyPlayers;
017
018public class GameQuickStart implements GameStartData {
019
020    private final Arena arena;
021    private final PlayersRefMap<PlayerStartData> playersData;
022
023    public GameQuickStart(DAC dac, Arena arena) {
024        this.arena = arena;
025        this.playersData = new PlayersRefHashMap<PlayerStartData>();
026
027        StartRegion startRegion = arena.getStartRegion();
028        Iterator<Color> colors = dac.getColors().iterator();
029        for (Player player : Bukkit.getOnlinePlayers()) {
030            if (startRegion.contains(player)) {
031                if (!colors.hasNext()) {
032                    throw new TooManyPlayers(dac.getMessages()
033                            .get("quickstart.toomanyplayers"));
034                }
035
036                Color color = colors.next();
037                PlayerStartData playerData = new SimplePlayerStartData(color, player);
038                playersData.put(player, playerData);
039            }
040        }
041    }
042
043    @Override
044    public Arena getArena() {
045        return arena;
046    }
047
048    @Override
049    public PlayersRefMap<? extends PlayerStartData> getPlayersData() {
050        return playersData;
051    }
052
053    @Override
054    public PlayersRefSet getSpectators() {
055        return new PlayersRefHashSet();
056    }
057
058    public int size() {
059        return playersData.size();
060    }
061}
